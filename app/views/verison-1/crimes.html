How it works

You can find the HTML for checkboxes component in the design system.

Copy the javascript and put it into your routes.js file.

Make sure the form action in the HTML and name of the function are the same. 

  

//Checkboxes

// Run this code when a form is submitted to 'about-children-answer'
router.post('/about-children-answer', function (req, res) {
// Make a variable and give it the value from 'howmanychildren'
var howmanychildren = req.session.data['nochildren']
// Check whether the variable matches a condition
if (howmanychildren == "no-children"){
  // Send user to next page
  res.redirect('/private-beta/v2/bereavement-support-payment/nochildren-not-eligible')
} else {
  // Send user to ineligible page
  res.redirect('/private-beta/v2/bereavement-support-payment/country')
}
})
